    <script>
        function get_len() {
            $.ajax({
                type: "GET",
                url: "/get_len",
                data: $('form').serialize(),
                success: function(response) {
                    var response = jQuery.parseJSON(response)
                    $('#temp').html(response.temp)
                },
                error: function(error) {
                    console.log(error);
                }
            });
        }
    </script>

    <form action="/get_len" method="get" name="form">
        <input type="button" value="Обновить данные" onclick="get_len();">
    </form>
